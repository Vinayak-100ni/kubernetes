## to go inside the pod 
   kubectl get pods -n mysql                             [for getting the name of the pod/container]
   kubectl exec -it mysql-statefulset-0 -n mysql -- bash 
                        |
            [container name / pod name]

## create base64 value for secret file
   echo "root" | base64

## to delete all pods of statefulset 
   kubectl delete statefulset mysql-statefulset -n mysql
                                    |
                            [statefulset name]
